üîë SSH Key Setup Guide 
# 1. Generate SSH key pair (if not exists)
ssh-keygen -t ed25519 -C "your_email@example.com" -f ~/.ssh/deployment_key

# 2. View your public key
cat ~/.ssh/deployment_key.pub

# 3. Add to server's authorized_keys
ssh-copy-id -i ~/.ssh/deployment_key.pub user@server

# Or manually:
ssh user@server "mkdir -p ~/.ssh && cat >> ~/.ssh/authorized_keys" < ~/.ssh/deployment_key.pub
