```bash
# Validation function example
validate_deployment_readiness() {
    # Check disk space
    disk_usage=$(df -P . | tail -1 | awk '{print $5}' | sed 's/%//')
    if [ "$disk_usage" -gt 80 ]; then
        echo "‚ùå Insufficient disk space: ${disk_usage}%"
        return 1
    fi

    # Check database connectivity
    php artisan migrate:status >/dev/null || {
        echo "‚ùå Database connection failed"
        return 1
    }

    # Check required services
    for service in nginx php8.2-fpm mysql; do
        if ! sudo systemctl is-active "$service" >/dev/null; then
            echo "‚ùå Service not running: $service"
            return 1
        fi
    done

    echo "‚úÖ Deployment readiness validated"
    return 0
}
```

**Release Retention Policy:**

```bash
# Automated release cleanup
cleanup_old_releases() {
    echo "üóëÔ∏è Cleaning up old releases..."

    # Keep last 5 releases
    KEEP_RELEASES=5

    # Count current releases
    release_count=$(ls -1 releases/ | grep "^20" | wc -l)

    if [ "$release_count" -gt "$KEEP_RELEASES" ]; then
        # Calculate how many to remove
        remove_count=$((release_count - KEEP_RELEASES))

        echo "üìä Found $release_count releases, removing oldest $remove_count"

        # Remove oldest releases
        ls -1 releases/ | grep "^20" | sort | head -n $remove_count | while read release; do
            echo "üóëÔ∏è Removing old release: $release"
            rm -rf "releases/$release"
        done

        echo "‚úÖ Release cleanup completed"
    else
        echo "‚úÖ Release count within limits ($release_count/$KEEP_RELEASES)"
    fi
}
```

```bash
optimize_database_performance() {

    echo "1Ô∏è‚É£ Index analysis..."
    mysql -u "$DB_USERNAME" -p"$DB_PASSWORD" "$DB_DATABASE" -e "
        SELECT
            table_name,

            sub_part,
            packed,
            nullable,
            index_type
        FROM information_schema.statistics
        WHERE table_schema = '$DB_DATABASE'

    " > database_indexes.txt

    echo "üìã Index analysis saved to: database_indexes.txt"

    # 2. Query performance analysis
    echo "2Ô∏è‚É£ Query performance check..."
    mysql -u "$DB_USERNAME" -p"$DB_PASSWORD" -e "
        SHOW VARIABLES LIKE 'slow_query_log';
        SHOW VARIABLES LIKE 'long_query_time';

    echo "3Ô∏è‚É£ Table optimization..."
    mysql -u "$DB_USERNAME" -p"$DB_PASSWORD" "$DB_DATABASE" -e "
        SELECT
            table_name,
            ROUND(((data_length + index_length) / 1024 / 1024), 2) AS 'Size (MB)',
            table_rows,
            ROUND(((data_length + index_length) / table_rows), 2) AS 'Avg Row Size'
        FROM information_schema.tables
        WHERE table_schema = '$DB_DATABASE'
        AND table_rows > 0
```

```bash
    --single-transaction \
    --quick \
    --lock-tables=false \
    -u "$DB_USERNAME" \
    -p"$DB_PASSWORD" \
    -h "$DB_HOST" \
    "$DB_DATABASE" > "$BACKUP_FILE"

# Compress backup
gzip "$BACKUP_FILE"

# Verify backup
if [ -f "${BACKUP_FILE}.gz" ]; then
    echo "‚úÖ Database backup completed: ${BACKUP_FILE}.gz"
else
    echo "‚ùå Database backup failed"
    exit 1
fi

# Cleanup old backups (keep last 7 days)
find "$BACKUP_DIR" -name "*.sql.gz" -mtime +7 -delete

echo "‚úÖ Database backup and cleanup completed"
EOF

    chmod +x database_backup.sh

    # 2. File backup
    echo "2Ô∏è‚É£ File backup configuration..."
    cat > file_backup.sh << 'EOF'
#!/bin/bash

BACKUP_DIR="backups/files"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# Backup critical files and directories

    --exclude='storage/logs/*' \
    --exclude='bootstrap/cache/*' \
    --exclude='node_modules' \
    --exclude='vendor' \
    --exclude='.git' \
    .

# Verify backup
if [ -f "$BACKUP_FILE" ]; then
    echo "‚úÖ File backup completed: $BACKUP_FILE"
else
    echo "‚ùå File backup failed"
    exit 1
fi

# Cleanup old backups (keep last 5 days)
find "$BACKUP_DIR" -name "*.tar.gz" -mtime +5 -delete

echo "‚úÖ File backup and cleanup completed"
EOF

    chmod +x file_backup.sh

    # 3. Configuration backup
    echo "3Ô∏è‚É£ Configuration backup..."
    cat > config_backup.sh << 'EOF'
#!/bin/bash

BACKUP_DIR="backups/config"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# Backup configuration files

    (crontab -l 2>/dev/null; echo "0 3 * * * $PWD/file_backup.sh >> backup.log 2>&1") | crontab -
```

```bash
#!/bin/bash

send_alert() {

    echo "$(date): [$level] $message" >> /var/log/system_alerts.log
```
