🔐 SSH Connection Details

🔑 SSH Key Setup Guide 
# 1. Generate SSH key pair (if not exists)
ssh-keygen -t ed25519 -C "your_email@example.com" -f ~/.ssh/deployment_key

# 2. View your public key
cat ~/.ssh/deployment_key.pub

# 3. Add to server's authorized_keys
ssh-copy-id -i ~/.ssh/deployment_key.pub user@server

# Or manually:
ssh user@server "mkdir -p ~/.ssh && cat >> ~/.ssh/authorized_keys" < ~/.ssh/deployment_key.pub
🌐 Server Environment
System information needed for deployment
📋 Commands to Get Server Info 
# Connect via SSH (using config)
ssh hostinger-factolo

# Or direct connection
ssh username@server.com -p 22

# Get OS information
cat /etc/os-release | grep -E "^(NAME|VERSION)"
# Or
lsb_release -a

# Check web server
nginx -v 2>/dev/null || apache2 -v 2>/dev/null || httpd -v 2>/dev/null

# Check PHP version
php -v | head -1

# Check Node.js version (if needed)
node -v 2>/dev/null || echo "Node.js not installed"
